    const popup = document.getElementById("popup");
    const popupCloseButton = document.getElementById("popupClose");

    const toast = document.getElementById("statusToast");
    const sections = document.querySelectorAll(".story-section");
    const translateButton = document.querySelector(".lang-btn");
    const translateButtonLabel = document.getElementById("translateButtonLabel");

    let toastTimer;
    let currentLanguage = "en";
    let translationReady = false;
    let ticking = false;
    const kannadaLabel = "ಕನ್ನಡದಲ್ಲಿ ಓದಿ";

    // Clean up duplicate/unnecessary sections
    document.querySelectorAll("main > section:not(.story-section)").forEach(s => s.remove());

    function showToast(message) {
      toast.textContent = message;
      toast.classList.add("show");
      window.clearTimeout(toastTimer);
      toastTimer = window.setTimeout(() => toast.classList.remove("show"), 2600);
    }

    function openPopup() {
      popup.classList.add("open");
      popup.setAttribute("aria-hidden", "false");
      document.body.classList.add("modal-open");
    }

    function closePopup() {
      popup.classList.remove("open");
      popup.setAttribute("aria-hidden", "true");
      document.body.classList.remove("modal-open");
    }

    function setTranslateButtonState(language) {
      const isKannada = language === "kn";
      translateButton.setAttribute("aria-pressed", isKannada ? "true" : "false");
      translateButtonLabel.textContent = isKannada ? "Read in English" : kannadaLabel;
    }

    function applyLanguage(language, attempt = 0) {
      const combo = document.querySelector(".goog-te-combo");

      if (!combo) {
        if (attempt < 24) {
          window.setTimeout(() => applyLanguage(language, attempt + 1), 250);
          return;
        }
        showToast("Translation is still loading. Please tap again in a moment.");
        return;
      }

      combo.value = language;
      combo.dispatchEvent(new Event("change"));
      currentLanguage = language;
      setTranslateButtonState(language);
      showToast(language === "kn" ? "Translating the page to Kannada." : "Switched back to English.");
    }

    function translatePage() {
      const nextLang = currentLanguage === "kn" ? "en" : "kn";
      if (!translationReady && !document.querySelector(".goog-te-combo")) {
        showToast("Preparing Kannada translation...");
      }
      applyLanguage(nextLang);
    }

    function updateParallax() {
      const vh = window.innerHeight || document.documentElement.clientHeight;

      sections.forEach((section) => {
        const rect = section.getBoundingClientRect();
        const offset = rect.top + (rect.height / 2) - (vh / 2);
        const progress = offset / vh;
        const reveal = Math.max(0, Math.min(1, (vh * 0.92 - rect.top) / (vh * 0.82)));
        const bgShift = Math.max(-40, Math.min(40, progress * -28));
        const tilt = Math.max(-4, Math.min(4, progress * -2.6));

        section.style.setProperty("--bg-shift", bgShift.toFixed(2));
        section.style.setProperty("--card-tilt", tilt.toFixed(2));
        section.style.setProperty("--reveal", reveal.toFixed(3));

        if (reveal > 0.16) section.classList.add("show");
      });

      ticking = false;
    }

    function requestParallaxUpdate() {
      if (!ticking) {
        window.requestAnimationFrame(updateParallax);
        ticking = true;
      }
    }

    window.googleTranslateElementInit = function() {
      new google.translate.TranslateElement({
        pageLanguage: "en",
        includedLanguages: "en,kn",
        autoDisplay: false,
        multilanguagePage: false
      }, "google_translate_element");
      translationReady = true;
    };

    // Event Listeners
    popup.addEventListener("click", (e) => { if(e.target === popup) closePopup(); });
    popupCloseButton.addEventListener("click", closePopup);
    
    enquiryForm.addEventListener("submit", (e) => {
      e.preventDefault();
      closePopup();
      enquiryForm.reset();
      showToast("Thanks! Your enquiry has been recorded.");
    });

    document.addEventListener("keydown", (e) => {
      if (e.key === "Escape" && popup.classList.contains("open")) closePopup();
    });

    window.addEventListener("scroll", requestParallaxUpdate, { passive: true });
    window.addEventListener("resize", requestParallaxUpdate);
    window.addEventListener("load", () => {
      requestParallaxUpdate();
      window.setTimeout(openPopup, 700);
    });

    requestParallaxUpdate();
    setTranslateButtonState(currentLanguage);
  </script>

    const popup = document.getElementById("popup");
    const popupCloseButton = document.getElementById("popupClose");
    const enquiryForm = document.getElementById("enquiryForm");
    const toast = document.getElementById("statusToast");
    const sections = document.querySelectorAll(".story-section");
    const translateButton = document.querySelector(".lang-btn");
    const translateButtonLabel = document.getElementById("translateButtonLabel");

    let toastTimer;
    let currentLanguage = "en";
    let translationReady = false;
    let ticking = false;
    const kannadaLabel = "ಕನ್ನಡದಲ್ಲಿ ಓದಿ";

    // Clean up duplicate/unnecessary sections
    document.querySelectorAll("main > section:not(.story-section)").forEach(s => s.remove());

    function showToast(message) {
      toast.textContent = message;
      toast.classList.add("show");
      window.clearTimeout(toastTimer);
      toastTimer = window.setTimeout(() => toast.classList.remove("show"), 2600);
    }

    function openPopup() {
      popup.classList.add("open");
      popup.setAttribute("aria-hidden", "false");
      document.body.classList.add("modal-open");
    }

    function closePopup() {
      popup.classList.remove("open");
      popup.setAttribute("aria-hidden", "true");
      document.body.classList.remove("modal-open");
    }

    function setTranslateButtonState(language) {
      const isKannada = language === "kn";
      translateButton.setAttribute("aria-pressed", isKannada ? "true" : "false");
      translateButtonLabel.textContent = isKannada ? "Read in English" : kannadaLabel;
    }

    function applyLanguage(language, attempt = 0) {
      const combo = document.querySelector(".goog-te-combo");

      if (!combo) {
        if (attempt < 24) {
          window.setTimeout(() => applyLanguage(language, attempt + 1), 250);
          return;
        }
        showToast("Translation is still loading. Please tap again in a moment.");
        return;
      }

      combo.value = language;
      combo.dispatchEvent(new Event("change"));
      currentLanguage = language;
      setTranslateButtonState(language);
      showToast(language === "kn" ? "Translating the page to Kannada." : "Switched back to English.");
    }

    function translatePage() {
      const nextLang = currentLanguage === "kn" ? "en" : "kn";
      if (!translationReady && !document.querySelector(".goog-te-combo")) {
        showToast("Preparing Kannada translation...");
      }
      applyLanguage(nextLang);
    }

    function updateParallax() {
      const vh = window.innerHeight || document.documentElement.clientHeight;

      sections.forEach((section) => {
        const rect = section.getBoundingClientRect();
        const offset = rect.top + (rect.height / 2) - (vh / 2);
        const progress = offset / vh;
        const reveal = Math.max(0, Math.min(1, (vh * 0.92 - rect.top) / (vh * 0.82)));
        const bgShift = Math.max(-40, Math.min(40, progress * -28));
        const tilt = Math.max(-4, Math.min(4, progress * -2.6));

        section.style.setProperty("--bg-shift", bgShift.toFixed(2));
        section.style.setProperty("--card-tilt", tilt.toFixed(2));
        section.style.setProperty("--reveal", reveal.toFixed(3));

        if (reveal > 0.16) section.classList.add("show");
      });

      ticking = false;
    }

    function requestParallaxUpdate() {
      if (!ticking) {
        window.requestAnimationFrame(updateParallax);
        ticking = true;
      }
    }

    window.googleTranslateElementInit = function() {
      new google.translate.TranslateElement({
        pageLanguage: "en",
        includedLanguages: "en,kn",
        autoDisplay: false,
        multilanguagePage: false
      }, "google_translate_element");
      translationReady = true;
    };

    // Event Listeners
    popup.addEventListener("click", (e) => { if(e.target === popup) closePopup(); });
    popupCloseButton.addEventListener("click", closePopup);
    
    enquiryForm.addEventListener("submit", (e) => {
      e.preventDefault();
      closePopup();
      enquiryForm.reset();
      showToast("Thanks! Your enquiry has been recorded.");
    });

    document.addEventListener("keydown", (e) => {
      if (e.key === "Escape" && popup.classList.contains("open")) closePopup();
    });

    window.addEventListener("scroll", requestParallaxUpdate, { passive: true });
    window.addEventListener("resize", requestParallaxUpdate);
    window.addEventListener("load", () => {
      requestParallaxUpdate();
      window.setTimeout(openPopup, 700);
    });

    requestParallaxUpdate();
    setTranslateButtonState(currentLanguage);
  </script>

    const popup = document.getElementById("popup");
    const popupCloseButton = document.getElementById("popupClose");
    const enquiryForm = document.getElementById("admissionForm");
    const cancelButton = document.getElementById("cancelButton");
    const toast = document.getElementById("statusToast");
    const sections = document.querySelectorAll(".story-section");
    const translateButton = document.querySelector(".lang-btn");
    const translateButtonLabel = document.getElementById("translateButtonLabel");
    const msgInput = document.getElementById('message');
    const charCounter = document.getElementById('charCount');
    const menuToggle = document.getElementById("menuToggle");
    const navMenu = document.querySelector("header nav");

    let toastTimer;
    let currentLanguage = "en";
    let translationReady = false;
    let ticking = false;
    const kannadaLabel = "ಕನ್ನಡದಲ್ಲಿ ಓದಿ";

    // Clean up duplicate/unnecessary sections
    document.querySelectorAll("main > section:not(.story-section)").forEach(s => s.remove());

    function showToast(message) {
      toast.textContent = message;
      toast.classList.add("show");
      window.clearTimeout(toastTimer);
      toastTimer = window.setTimeout(() => toast.classList.remove("show"), 2600);
    }

    function openPopup() {
      if (enquiryForm) {
        enquiryForm.reset();
        if (charCounter) charCounter.textContent = `0 / 1000`;
      }
      popup.classList.add("open");
      popup.setAttribute("aria-hidden", "false");
      document.body.classList.add("modal-open");
    }

    function closePopup() {
      if (enquiryForm) {
        enquiryForm.reset();
      }
      popup.classList.remove("open");
      popup.setAttribute("aria-hidden", "true");
      document.body.classList.remove("modal-open");
    }

    // Mobile Menu Toggle
    if (menuToggle && navMenu) {
      menuToggle.addEventListener("click", () => {
        navMenu.classList.toggle("active");
        menuToggle.classList.toggle("active");
      });
      navMenu.querySelectorAll("a").forEach(link => {
        link.addEventListener("click", () => {
          navMenu.classList.remove("active");
          menuToggle.classList.remove("active");
        });
      });
    }

    const MAX_CHARS = 1000;
    if (msgInput && charCounter) {
      msgInput.addEventListener('input', (e) => {
        const len = e.target.value.length;
        charCounter.textContent = `${len} / ${MAX_CHARS}`;
        charCounter.style.color = len >= MAX_CHARS ? 'var(--f-error)' : 'var(--f-muted)';
      });
    }

    function setTranslateButtonState(language) {
      const isKannada = language === "kn";
      translateButton.setAttribute("aria-pressed", isKannada ? "true" : "false");
      translateButtonLabel.textContent = isKannada ? "Read in English" : kannadaLabel;
    }

    function applyLanguage(language, attempt = 0) {
      const combo = document.querySelector(".goog-te-combo");

      if (!combo) {
        if (attempt < 24) {
          window.setTimeout(() => applyLanguage(language, attempt + 1), 250);
          return;
        }
        showToast("Translation is still loading. Please tap again in a moment.");
        return;
      }

      combo.value = language;
      combo.dispatchEvent(new Event("change"));
      currentLanguage = language;
      setTranslateButtonState(language);
      showToast(language === "kn" ? "Translating the page to Kannada." : "Switched back to English.");
    }

    function translatePage() {
      const nextLang = currentLanguage === "kn" ? "en" : "kn";
      if (!translationReady && !document.querySelector(".goog-te-combo")) {
        showToast("Preparing Kannada translation...");
      }
      applyLanguage(nextLang);
    }

    function updateParallax() {
      const vh = window.innerHeight || document.documentElement.clientHeight;

      sections.forEach((section) => {
        const rect = section.getBoundingClientRect();
        const offset = rect.top + (rect.height / 2) - (vh / 2);
        const progress = offset / vh;
        const reveal = Math.max(0, Math.min(1, (vh * 0.92 - rect.top) / (vh * 0.82)));
        const bgShift = Math.max(-40, Math.min(40, progress * -28));
        const tilt = Math.max(-4, Math.min(4, progress * -2.6));

        section.style.setProperty("--bg-shift", bgShift.toFixed(2));
        section.style.setProperty("--card-tilt", tilt.toFixed(2));
        section.style.setProperty("--reveal", reveal.toFixed(3));

        if (reveal > 0.16) section.classList.add("show");
      });

      ticking = false;
    }

    function requestParallaxUpdate() {
      if (!ticking) {
        window.requestAnimationFrame(updateParallax);
        ticking = true;
      }
    }

    window.googleTranslateElementInit = function() {
      new google.translate.TranslateElement({
        pageLanguage: "en",
        includedLanguages: "en,kn",
        autoDisplay: false,
        multilanguagePage: false
      }, "google_translate_element");
      translationReady = true;
    };

    // Event Listeners
    popup.addEventListener("click", (e) => { if(e.target === popup) closePopup(); });
    popupCloseButton.addEventListener("click", closePopup);

    if (enquiryForm) {
      enquiryForm.addEventListener("submit", (e) => {
        e.preventDefault();
        closePopup();
        showToast("Thanks! Your enquiry has been recorded.");
      });
    }

    if (cancelButton) {
      cancelButton.addEventListener("click", () => {
        if(confirm("Discard enquiry? / ವಿಚಾರಣೆಯನ್ನು ರದ್ದುಗೊಳಿಸಬೇಕೆ?")) {
          closePopup();
        }
      });
    }

    document.addEventListener("keydown", (e) => {
      if (e.key === "Escape" && popup.classList.contains("open")) closePopup();
    });

    window.addEventListener("scroll", requestParallaxUpdate, { passive: true });
    window.addEventListener("resize", requestParallaxUpdate);
    window.addEventListener("load", () => {
      requestParallaxUpdate();
      window.setTimeout(openPopup, 700);
    });

    requestParallaxUpdate();
    setTranslateButtonState(currentLanguage);
  </script>

    // Application Interface References
    const SchoolApp = {
      ui: {
        modal: document.querySelector("#popup"),
        closer: document.querySelector("#popupClose"),
        admissionForm: document.querySelector("#admissionForm"),
        abortBtn: document.querySelector("#cancelButton"),
        notifier: document.querySelector("#statusToast"),
        contentBlocks: document.querySelectorAll(".story-section"),
        messageArea: document.querySelector("#message"),
        charDisplay: document.querySelector("#charCount"),
        navSwitcher: document.querySelector("#menuToggle"),
        navigation: document.querySelector("header nav")
      },
      state: {
        timer: null,
        isAnimating: false
      }
    };

    // Clean up duplicate/unnecessary sections
    document.querySelectorAll("main > section:not(.story-section)").forEach(s => s.remove());

    function triggerNotification(text) {
      SchoolApp.ui.notifier.textContent = text;
      SchoolApp.ui.notifier.classList.add("show");
      window.clearTimeout(SchoolApp.state.timer);
      SchoolApp.state.timer = window.setTimeout(() => SchoolApp.ui.notifier.classList.remove("show"), 2600);
    }

    function openPopup() {
      if (SchoolApp.ui.admissionForm) {
        SchoolApp.ui.admissionForm.reset();
        if (SchoolApp.ui.charDisplay) SchoolApp.ui.charDisplay.textContent = `0 / 1000`;
      }
      SchoolApp.ui.modal.classList.add("open");
      SchoolApp.ui.modal.setAttribute("aria-hidden", "false");
      document.body.classList.add("modal-open");
    }

    function closePopup() {
      if (SchoolApp.ui.admissionForm) {
        SchoolApp.ui.admissionForm.reset();
      }
      SchoolApp.ui.modal.classList.remove("open");
      SchoolApp.ui.modal.setAttribute("aria-hidden", "true");
      document.body.classList.remove("modal-open");
    }

    // Mobile Menu Toggle
    if (SchoolApp.ui.navSwitcher && SchoolApp.ui.navigation) {
      SchoolApp.ui.navSwitcher.addEventListener("click", () => {
        SchoolApp.ui.navigation.classList.toggle("active");
        SchoolApp.ui.navSwitcher.classList.toggle("active");
      });
      SchoolApp.ui.navigation.querySelectorAll("a").forEach(link => {
        link.addEventListener("click", () => {
          SchoolApp.ui.navigation.classList.remove("active");
          SchoolApp.ui.navSwitcher.classList.remove("active");
        });
      });
    }

    const MAX_CHARS = 1000;
    if (SchoolApp.ui.messageArea && SchoolApp.ui.charDisplay) {
      SchoolApp.ui.messageArea.addEventListener('input', (e) => {
        const len = e.target.value.length;
        SchoolApp.ui.charDisplay.textContent = `${len} / ${MAX_CHARS}`;
        SchoolApp.ui.charDisplay.style.color = len >= MAX_CHARS ? 'var(--f-error)' : 'var(--f-muted)';
      });
    }

    function refreshScrollVisuals() {
      const vh = window.innerHeight || document.documentElement.clientHeight;

      SchoolApp.ui.contentBlocks.forEach((section) => {
        const rect = section.getBoundingClientRect();
        const offset = rect.top + (rect.height / 2) - (vh / 2);
        const progress = offset / vh;
        const reveal = Math.max(0, Math.min(1, (vh * 0.92 - rect.top) / (vh * 0.82)));
        const bgShift = Math.max(-40, Math.min(40, progress * -28));
        const tilt = Math.max(-4, Math.min(4, progress * -2.6));

        section.style.setProperty("--bg-shift", bgShift.toFixed(2));
        section.style.setProperty("--card-tilt", tilt.toFixed(2));
        section.style.setProperty("--reveal", reveal.toFixed(3));

        if (reveal > 0.16) section.classList.add("show");
      });

      SchoolApp.state.isAnimating = false;
    }

    function requestScrollRefresh() {
      if (!SchoolApp.state.isAnimating) {
        window.requestAnimationFrame(refreshScrollVisuals);
        SchoolApp.state.isAnimating = true;
      }
    }

    // Event Listeners
    SchoolApp.ui.modal.addEventListener("click", (e) => { if(e.target === SchoolApp.ui.modal) closePopup(); });
    SchoolApp.ui.closer.addEventListener("click", closePopup);

    if (SchoolApp.ui.admissionForm) {
      SchoolApp.ui.admissionForm.addEventListener("submit", (e) => {
        e.preventDefault();
        closePopup();
        triggerNotification("Thanks! Your enquiry has been recorded.");
      });
    }

    if (SchoolApp.ui.abortBtn) {
      SchoolApp.ui.abortBtn.addEventListener("click", () => {
        if(confirm("Discard enquiry? / ವಿಚಾರಣೆಯನ್ನು ರದ್ದುಗೊಳಿಸಬೇಕೆ?")) {
          closePopup();
        }
      });
    }

    document.addEventListener("keydown", (e) => {
      if (e.key === "Escape" && SchoolApp.ui.modal.classList.contains("open")) closePopup();
    });

    window.addEventListener("scroll", requestScrollRefresh, { passive: true });
    window.addEventListener("resize", requestScrollRefresh);
    window.addEventListener("load", () => {
      requestScrollRefresh();
      window.setTimeout(openPopup, 700);
    });
  </script>
</body>
